## Reverse String

```swift
// swift
class Solution {
    func reverseString(_ s: inout [Character]) {
        helper(0, s.count - 1, &s)
    }
    func helper(_ l:Int, _ r: Int, _ s: inout [Character]) {
        if l >= r {return}
        let tmp = s[l]
        s[l] = s[r]
        s[r] = tmp
        helper(l+1, r-1, &s)
    }
}
```
