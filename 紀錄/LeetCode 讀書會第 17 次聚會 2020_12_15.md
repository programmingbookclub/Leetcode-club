class Solution {
    func hasCycle(_ head: ListNode?) -> Bool {
        var p1 = head
        var p2 = head
        while true {
            p1 = p1?.next
            p2 = p2?.next?.next
            if p2 == nil {return false}
            if p1 === p2 {return true}
        }
    }
}
